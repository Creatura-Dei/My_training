// 3 ( \n - один спецсимвол ) length - свойство, а не функция, поэтому добавлять скобки не нужно (str.length) __________________________________________________________________________________ ДОСТУП К СИМВОЛАМ Получение символа с позиции pos, можно с помощью [pos] ИЛИ Метод - str.at(pos) ВАЖНО: первый символ нулевой let str = 'Hello'; // Первый символ alert( str[0] ); // H - ОТРИЦАТЕЛЬНЫЕ ВОЗВР undefined alert( str.at(0) ); // H alert( str[str.length - 1] ); // o alert( str.at(-1) ); // o, если -2, то символ перед ним В ИТОГЕ: .at(pos) - преимущественнее, так как если отрицательное число, то отсчёт ведётся от конца строки let str = "Hello"; alert( str[-2] ); // undefined alert( str.at(-2) ); // 1 ПОСИМВОЛЬНЫЙ ПЕРЕБОР - for..of for (let char of "Hello") { alert(char); // H e l l o } __________________________________________________________________________________ СТРОКИ НЕИЗМЕНЯЕМЫ НЕЛЬЗЯ заменить символ посередине слова. __________________________________________________________________________________ ИЗМЕНЕНИЕ РЕГИСТРА alert( 'Interface'.toUpperCase() ); // INTERFACE alert( 'Interface'.toLowerCase() ); // interface __________________________________________________________________________________ ПОИСК ПОДСТРОКИ 2 метода: 1) str.indexOf(substr, pos) Он ищет подстроку substr в строке str, начиная с позиции pos, и возвращает позицию, на которой располагается совпадене, либо -1 при отсутствии совпадений let str = 'Widget with id'; alert( str.indexOf('Widget') ); // 0, потому что подстрока 'Widget' найдена
